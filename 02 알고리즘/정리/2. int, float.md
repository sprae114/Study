def gcd_euclidean(a, b):
    a, b = max(a, b), min(a, b)
    
    while b != 0:
        a, b = b, a % b
    
    return a

def solution(n):
    a = [False,False] + [True]*(n-1)
    answer = 0
    for i in range(2,n+1):
        if a[i]:
            answer += 1
            
            for j in range(2*i, n+1, i):
                a[j] = False

    return answer

print(solution(10))
